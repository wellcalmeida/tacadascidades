    <script>
        // Substitua pela URL do seu CSV publicado no Google Sheets
        const SHEET_URL = 'https://docs.google.com/spreadsheets/d/1lF1Yxm89ZScD11FcVjdVmCmpHOAYO4l5WOge4kIWsME/edit?gid=0#gid=0';
        const META_PONTOS = 1000; // Defina o valor máximo da régua

        async function atualizarPontos() {
            try {
                const response = await fetch(SHEET_URL);
                const data = await response.text();
                const linhas = data.split('\n');
                
                // Pega os valores das células B2 e B3 (ajuste conforme sua planilha)
                const pontosNinive = parseInt(linhas[1].split(',')[1]);
                const pontosJuda = parseInt(linhas[2].split(',')[1]);

                // Atualiza o texto
                document.getElementById('val-ninive').innerText = pontosNinive;
                document.getElementById('val-juda').innerText = pontosJuda;

                // Atualiza a altura das réguas (porcentagem)
                document.getElementById('ninive-fill').style.height = (pontosNinive / META_PONTOS * 100) + '%';
                document.getElementById('juda-fill').style.height = (pontosJuda / META_PONTOS * 100) + '%';
                
            } catch (error) {
                console.error("Erro ao buscar dados:", error);
            }
        }

        // Atualiza a cada 30 segundos
        setInterval(atualizarPontos, 30000);
        atualizarPontos();
    </script>
